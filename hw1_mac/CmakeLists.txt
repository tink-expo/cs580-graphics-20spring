cmake_minimum_required(VERSION 3.0.0)
project (raytracing)

file(GLOB SOURCES "*.cpp")
file(GLOB HEADERS "*.h")

set ( CMAKE_CXX_COMPILER "clang++" )

find_package(OpenGL REQUIRED)
set(GLUT_LIBRARIES libglut.3.dylib)
set(GLUT_INCLUDE_DIRS /opt/X11/include)
set(GLUT_LIBRARY_DIRS /opt/X11/lib)
set(GLM_INCLUDE_DIRS /usr/local/include/glm)
find_package(glfw3 CONFIG REQUIRED)
find_package(glm CONFIG REQUIRED)


add_compile_options ( "-O2" )
add_compile_definitions( "GL_SILENCE_DEPRECATION" )

add_executable(raytracing ${SOURCES} ${HEADERS})

include_directories ( ${OPENGL_INCLUDE_DIRS} ${GLUT_INCLUDE_DIRS} ${GLM_INCLUDE_DIRS})

link_libraries( ${OPENGL_LIBRARIES} ${GLUT_LIBRARIES} glfw )

link_directories ( ${OPENGL_LIBRARY_DIRS} ${GLUT_LIBRARY_DIRS})
